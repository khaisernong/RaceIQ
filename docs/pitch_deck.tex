% Problem
\begin{frame}{The Problem}
  \begin{itemize}
    \item Pit stop timing in endurance racing is decisive yet often based on gut feel.
    \item Pitting too early: lost track position. Too late: lap time loss from tire degradation.
    \item Teams need real-time, data-driven strategy — not post-race analysis.
  \end{itemize}
\end{frame}

% Solution
\begin{frame}{Our Solution: RaceIQ}
  \begin{itemize}
    \item AI-powered platform that transforms raw race data into pit stop recommendations.
    \item Live dashboard built with \textbf{Python + Streamlit} and \textbf{Plotly}.
    \item Tire degradation model learns track-specific wear patterns.
    \item Clear GO / MONITOR / PIT-NOW guidance for fast decisions.
  \end{itemize}
\end{frame}

% Architecture
\begin{frame}{Architecture}
  \begin{block}{Four Layers}
    \begin{enumerate}
      \item \textbf{Data Ingestion}: Flexible CSV loader with auto-detection, fallbacks.
      \item \textbf{Analytics Engine}: Feature engineering, consistency metrics, rolling stats.
      \item \textbf{Decision Support}: Tire degradation model, pit window logic.
      \item \textbf{Presentation}: Streamlit dashboard (4 tabs) with Plotly visuals.
    \end{enumerate}
  \end{block}
\end{frame}

% Model
\begin{frame}{Tire Degradation Model}
  \begin{itemize}
    \item Algorithm: \textbf{Polynomial regression} (degree 2) via scikit-learn.
    \item Features: \(\text{tire\_age},\, \text{tire\_age}^2,\, \text{air\_temp},\, \text{track\_temp}\).
    \item Performance observed: \(R^2\) = 0.17–0.27, RMSE = 1.7–3.7s (varies by track).
    \item Output used for: lap time forecasting, degradation rate, optimal pit lap.
  \end{itemize}
\end{frame}

% Results examples
\begin{frame}{Key Results (Examples)}
  \begin{itemize}

    \item \textbf{COTA} (with sector fallback): model runs and produces recommendations.
    \item Clear degradation trends: lap times rise with tire age; optimizer identifies pit window.
  \end{itemize}
\end{frame}
